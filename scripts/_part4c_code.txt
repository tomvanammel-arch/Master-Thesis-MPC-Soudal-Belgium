# This cell only loads CSV/JSON + reloads the viewer module.
# It does NOT rerun the expensive simulations.

from pathlib import Path
import sys

import pandas as pd

PROJECT_ROOT = Path("..").resolve()
SRC_DIR = PROJECT_ROOT / "src"
if str(SRC_DIR) not in sys.path:
    sys.path.insert(0, str(SRC_DIR))

import importlib

import online_MPC_1_HP_scenario_analysis as _hp_scen
import notebook_visualisation.nb10_hp_part34_viewer as _viewer

importlib.reload(_hp_scen)
importlib.reload(_viewer)

NOTEBOOKS_OUTPUT_DIR = PROJECT_ROOT / "output" / "notebooks"

# --- knobs: pick one scenario from Part 4A ---
SCENARIO_SELECT = 16  # must match `scenario_id` in SCENARIOS
# Optional explicit stem; if None, pick the newest CSV matching the id tag.
SCENARIO_NAME_STEM: str | None = None  # e.g. "part2_reference"; None -> glob

id_tag = _hp_scen.scenario_id_tag_for_filename(SCENARIO_SELECT)
if SCENARIO_NAME_STEM:
    safe = _hp_scen.sanitize_scenario_name_for_filename(SCENARIO_NAME_STEM)
    RESULTS_15MIN_PATH = NOTEBOOKS_OUTPUT_DIR / f"online_hp_15min_notebook_10_scenario_{id_tag}_{safe}.csv"
else:
    matches = sorted(
        NOTEBOOKS_OUTPUT_DIR.glob(f"online_hp_15min_notebook_10_scenario_{id_tag}_*.csv"),
        key=lambda p: p.stat().st_mtime,
    )
    if not matches:
        raise FileNotFoundError(
            f"No scenario CSV found for id tag {id_tag!r} under {NOTEBOOKS_OUTPUT_DIR}"
        )
    RESULTS_15MIN_PATH = matches[-1]

_stem = RESULTS_15MIN_PATH.stem
if _stem.startswith("online_hp_15min_notebook_10_scenario_"):
    _summ_stem = "online_hp_summary_notebook_10_scenario_" + _stem[len("online_hp_15min_notebook_10_scenario_"):]
    summ_path = RESULTS_15MIN_PATH.with_name(_summ_stem + ".json")
else:
    summ_path = RESULTS_15MIN_PATH.with_suffix(".json")

print("Loading:", RESULTS_15MIN_PATH)
res_hp_online = pd.read_csv(RESULTS_15MIN_PATH)
summ_hp_online = None
if summ_path.exists():
    import json

    with open(summ_path, "r", encoding="utf-8") as f:
        summ_hp_online = json.load(f)
    print("Loaded summary JSON:", summ_path)
else:
    print("NOTE: summary JSON not found (optional):", summ_path)

_viewer.run_notebook10_part31_optimized_volumes(res_hp_online, PROJECT_ROOT)
_viewer.run_notebook10_part32_billing_comparison(res_hp_online, PROJECT_ROOT)
